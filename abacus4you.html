🔗 Google Sheets Integration

Results sync to your Google Sheet in real-time via Apps Script. Each entry stores: name, class, score, timing, accuracy, type, and timestamp.

Setup: Deploy the Apps Script below as a Web App, then add the URL to Admin → Settings → Webhook URL.

// Google Apps Script (Code.gs)
function doPost(e) {
  const data = JSON.parse(e.postData.contents);
  const sheet = SpreadsheetApp.openById(SHEET_ID)
    .getSheetByName('Competition');
  sheet.appendRow([
    new Date(), data.name, data.class,
    data.score, data.time, data.accuracy,
    data.type, data.questions
  ]);
  // Auto-sort by score
  const range = sheet.getDataRange();
  range.sort({column:4, ascending:false});
  return ContentService
    .createTextOutput(JSON.stringify({status:'ok'}))
    .setMimeType(ContentService.MimeType.JSON);
}

    // ─── DAILY RESET TRIGGER ───
function setupDailyReset() {
  ScriptApp.newTrigger('resetDailyCompetition')
    .timeBased().everyDays(1).atHour(0).create();
}
function resetDailyCompetition() {
  const ss = SpreadsheetApp.openById(SHEET_ID);
  const sheet = ss.getSheetByName('Daily');
  // Keep header, clear data rows
  if (sheet.getLastRow() > 1) {
    sheet.deleteRows(2, sheet.getLastRow() - 1);
  }
  // Log the reset
  ss.getSheetByName('Log').appendRow([new Date(), 'Daily reset completed']);
}

// ─── SEND EMAIL DIGEST ───
function sendDailyDigest() {
  const sheet = SpreadsheetApp.openById(SHEET_ID)
    .getSheetByName('Competition');
  const data = sheet.getDataRange().getValues();
  const top3 = data.slice(1,4);
  let html = '<h2>Today\'s Top Performers</h2>';
  top3.forEach((r,i) => {
    html += \`<p>#\${i+1}: <b>\${r[1]}</b> — Score: \${r[3]}</p>\`;
  });
  GmailApp.sendEmail(ADMIN_EMAIL, 'Daily Math Results', '', {htmlBody: html});
}

// ─── REAL-TIME LEADERBOARD ───
function doGet() {
  const sheet = SpreadsheetApp.openById(SHEET_ID)
    .getSheetByName('Competition');
  const data = sheet.getDataRange().getValues();
  const rows = data.slice(1).map(r => ({
    name:r[1], class:r[2], score:r[3],
    time:r[4], accuracy:r[5], type:r[6]
  }));
  return ContentService
    .createTextOutput(JSON.stringify(rows))
    .setMimeType(ContentService.MimeType.JSON);
}
